# 系统数据流设计

## 📋 设计概述

### 设计目标
设计一个高效、可靠、可扩展的数据流系统，支持外盘黄金量化交易信号系统的全流程数据处理。

### 设计原则
1. **数据一致性**: 确保数据在各个环节的一致性
2. **实时性**: 支持实时数据处理和推送
3. **可靠性**: 数据不丢失，可追溯
4. **扩展性**: 支持多数据源，易于扩展
5. **性能**: 优化数据处理性能，减少延迟

class DataPrefetcher:
    def __init__(self):
        self.cache = {}
        self.prefetch_queue = []
    
    def prefetch(self, symbol, timeframe, start_date, end_date):
        """预加载数据到缓存"""
        cache_key = f"{symbol}:{timeframe}:{start_date}:{end_date}"
        if cache_key not in self.cache:
            # 异步加载数据
            self._async_load_data(cache_key, symbol, timeframe, start_date, end_date)
    
    def get_data(self, symbol, timeframe, start_date, end_date):
        """获取数据，优先从缓存读取"""
        cache_key = f"{symbol}:{timeframe}:{start_date}:{end_date}"
        if cache_key in self.cache:
            return self.cache[cache_key]
        else:
            # 同步加载
            data = fetch_data(symbol, timeframe, start_date, end_date)
            self.cache[cache_key] = data
            return data
```

from concurrent.futures import ThreadPoolExecutor, as_completed

def calculate_indicators_parallel(data, indicators):
    """并行计算多个技术指标"""
    results = {}
    
    with ThreadPoolExecutor(max_workers=4) as executor:
        # 提交计算任务
        future_to_indicator = {
            executor.submit(calculate_indicator, data, indicator): indicator
            for indicator in indicators
        }
        
        # 收集结果
        for future in as_completed(future_to_indicator):
            indicator = future_to_indicator[future]
            try:
                results[indicator] = future.result()
            except Exception as e:
                logger.error(f"计算指标 {indicator} 失败: {e}")
    
    return results
```

class IncrementalCalculator:
    def __init__(self):
        self.previous_values = {}
    
    def calculate_sma_incremental(self, new_price, period=20):
        """增量计算移动平均线"""
        if 'sma_values' not in self.previous_values:
            self.previous_values['sma_values'] = []
        
        values = self.previous_values['sma_values']
        values.append(new_price)
        
        if len(values) > period:
            values.pop(0)  # 移除最旧的值
        
        if len(values) == period:
            sma = sum(values) / period
            return sma
        else:
            return None  # 数据不足
```

### 3. 数据写入优化

#### 批量写入
```python
def batch_insert_data(connection, data_records, batch_size=1000):
    """批量插入数据，提高写入性能"""
    cursor = connection.cursor()
    
    for i in range(0, len(data_records), batch_size):
        batch = data_records[i:i+batch_size]
        
        # 构建批量插入SQL
        values_placeholders = []
        values_params = []
        
        for record in batch:
            placeholders = []
            for field in ['symbol', 'timeframe', 'timestamp', 'open', 'high', 'low', 'close', 'volume']:
                placeholders.append('%s')
                values_params.append(record[field])
            values_placeholders.append(f"({','.join(placeholders)})")
        
        sql = f"""
        INSERT INTO historical_data 
        (symbol, timeframe, timestamp, open, high, low, close, volume)
        VALUES {','.join(values_placeholders)}
        ON CONFLICT (symbol, timeframe, timestamp) DO NOTHING
        """
        
        cursor.execute(sql, values_params)
    
    connection.commit()
    cursor.close()
```

#### 异步写入
```python
import asyncio
from aioredis import Redis

async def async_write_to_redis(redis_client, stream_key, data):
    """异步写入Redis Stream"""
    try:
        # 异步写入
        await redis_client.xadd(stream_key, data)
        return True
    except Exception as e:
        logger.error(f"写入Redis失败: {e}")
        return False

async def write_data_async(data_list):
    """批量异步写入数据"""
    redis_client = await Redis.from_url("redis://localhost")
    
    tasks = []
    for data in data_list:
        task = asyncio.create_task(async_write_to_redis(redis_client, data['stream'], data['values']))
        tasks.append(task)
    
    # 等待所有写入完成
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    await redis_client.close()
    return results
```

### 1. 数据完整性监控
```python
class DataQualityMonitor:
    def __init__(self):
        self.metrics = {}
    
    def check_data_completeness(self, data, expected_interval='1m'):
        """检查数据完整性"""
        if len(data) < 2:
            return {'status': 'insufficient_data', 'missing_rate': 1.0}
        
        # 计算时间间隔
        time_diffs = []
        for i in range(1, len(data)):
            diff = (data[i]['timestamp'] - data[i-1]['timestamp']).total_seconds()
            time_diffs.append(diff)
        
        # 统计缺失数据点
        expected_diff = self._get_expected_seconds(expected_interval)
        missing_count = sum(1 for diff in time_diffs if diff > expected_diff * 1.5)
        
        missing_rate = missing_count / len(time_diffs) if time_diffs else 0
        
        return {
            'status': 'good' if missing_rate < 0.05 else 'warning',
            'missing_rate': missing_rate,
            'total_points': len(data),
            'missing_points': missing_count
        }
    
    def _get_expected_seconds(self, interval):
        """获取预期时间间隔（秒）"""

            '1
