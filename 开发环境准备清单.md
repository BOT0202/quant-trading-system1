### 5. 开发环境验证

#### 环境验证脚本
```python
# check_environment.py
import sys
import subprocess

def check_python():
    print("检查Python环境...")
    version = sys.version_info
    if version.major == 3 and version.minor >= 9:
        print(f"✅ Python版本: {sys.version}")
        return True
    else:
        print(f"❌ Python版本过低: {sys.version}")
        return False

def check_packages():
    print("\n检查Python包...")
    packages = [
        ('backtrader', '2.0.0'),
        ('django', '4.2.0'),
        ('pandas', '1.5.0'),
        ('celery', '5.2.7'),
        ('redis', '4.5.0')
    ]
    
    all_ok = True
    for package, min_version in packages:
        try:
            module = __import__(package)
            version = getattr(module, '__version__', '未知')
            if version >= min_version:
                print(f"✅ {package}: {version}")
            else:
                print(f"⚠️  {package}: {version} (需要 {min_version}+)")
                all_ok = False
        except ImportError:
            print(f"❌ {package}: 未安装")
            all_ok = False
    
    return all_ok

def check_database():
    print("\n检查数据库连接...")
    try:
        import psycopg2
        conn = psycopg2.connect(
            host="localhost",
            database="goldquant_dev",
            user="goldquant",
            password="your_password"
        )
        conn.close()
        print("✅ PostgreSQL连接成功")
        return True
    except Exception as e:
        print(f"❌ PostgreSQL连接失败: {e}")
        return False

def check_redis():
    print("\n检查Redis连接...")
    try:
        import redis
        r = redis.Redis(host='localhost', port=6379, password='your_password')
        r.ping()
        print("✅ Redis连接成功")
        return True
    except Exception as e:
        print(f"❌ Redis连接失败: {e}")
        return False

def main():
    print("=" * 50)
    print("开发环境验证")
    print("=" * 50)
    
    results = []
    results.append(check_python())
    results.append(check_packages())
    results.append(check_database())
    results.append(check_redis())
    
    print("\n" + "=" * 50)
    if all(results):
        print("✅ 所有环境检查通过!")
    else:
        print("❌ 环境检查未通过，请修复问题")
    
    return all(results)

if __name__ == "__main__":
    success = main()
    sys.exit(0 if success else 1)
```
